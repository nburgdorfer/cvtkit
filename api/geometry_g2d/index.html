This module contains the following functions:

  • essential_from_features(src_image_file, tgt_image_file, K) - Computes the essential matrix between two images based on features.
  • fundamental_from_KP(K, P_src, P_tgt) - Computes the fundamental matrix between two images from camera matrices.

  • visibility_map(src_depth, src_cam, depth_files, cam_files, src_index, pixel_th) - Computes a visibility map between each the pixels in the source depth map and each depth map in the provided list of target depth map files.

def essential_from_features(src_image_file: str, tgt_image_file: str, K: np.ndarray) -> np.ndarray:
    """Computes the essential matrix between two images based on features.

    Parameters:
        src_image_file: Input source image file
        tgt_image_file: Input target image file

    Returns:
        The essential matrix between the two images.
    """
    src_image = cv2.imread(src_image_file)
    tgt_image = cv2.imread(tgt_image_file)

    # compute matching features
    (src_points, tgt_points) = match_features(src_image, tgt_image)

    # Compute essential matrix
    E, mask = cv2.findEssentialMat(src_points, tgt_points, K, method=cv2.RANSAC)

    return E

def fundamental_from_KP(K: np.ndarray, P_src: np.ndarray, P_tgt: np.ndarray) -> np.ndarray :
    """Computes the fundamental matrix between two images from camera matrices.

    Parameters:
        K: Intrinsics matrix (3x3) for the two camera views.
        P_src: Extrinsics matrix (4x4) for the source camera.
        P_tgt: extrinsics matrix (4x4) for the target camera.

    Returns:
        The fundamental matrix between the two cameras.
    """
    R1 = P1[0:3,0:3]
    t1 = P1[0:3,3]
    R2 = P2[0:3,0:3]
    t2 = P2[0:3,3]

    t1aug = np.array([t1[0], t1[1], t1[2], 1])
    epi2 = np.matmul(P2,t1aug)
    epi2 = np.matmul(K,epi2[0:3])
    print('epipole 2: {} {}'.format(epi2[0]/epi2[2],epi2[1]/epi2[2]))

    R = np.matmul(R2,np.transpose(R1))
    t= t2- np.matmul(R,t1)
    K1inv = np.linalg.inv(K)
    K2invT = np.transpose(K1inv)
    tx = np.array([[0, -t[2], t[1]], [t[2], 0, -t[0]], [-t[1], t[0], 0]])
    F = np.matmul(K2invT,np.matmul(tx,np.matmul(R,K1inv)))
    F = F/np.amax(F)
    return F

def geometric_consistency_mask(src_depth: np.ndarray, src_cam: np.ndarray, tgt_depth: np.ndarray, tgt_cam: np.ndarray, pixel_th: float) -> np.ndarray:
    """Computes a geometric consistency mask between two depth maps.

    Parameters:
        src_depth: Source depth map for which the mask will be computed.
        src_cam: Source view camera parameters.
        tgt_depth: Target depth map used for geometric consistency checking.
        tgt_cam: Target view camera parameters.
        pixel_th: Pixel threshold used for binary consensus decisions.

    Returns:
        Binary geometric consistency mask for the source depth map.
    """
    height, width = src_depth.shape
    x_src, y_src = np.meshgrid(np.arange(0, width), np.arange(0, height))

    depth_reprojected, x2d_reprojected, y2d_reprojected, x2d_tgt, y2d_tgt = reproject(src_depth, src_cam, tgt_depth, tgt_cam)

    dist = np.sqrt((x2d_reprojected - x_src) ** 2 + (y2d_reprojected - y_src) ** 2)
    mask = np.where(dist < pixel_th, 1, 0)

    return mask

def homography_from_features(src_image_file: str, tgt_image_file: str) -> np.ndarray:
    """Computes a homography transformation between two images.

    Parameters:
        src_image_file: Input source image file.
        tgt_image_file: Input target image file.

    Returns:
        A homography matrix (3x3) that warps the target image to the source.
    """
    src_img = cv2.imread(src_image_file)
    tgt_img = cv2.imread(tgt_image_file)

    (height, width, _) = src_img.shape

    (src_points, tgt_points) = match_features(src_img, tgt_img)

    H, mask = cv2.findHomography(src_points, tgt_points, method=cv2.RANSAC)

    return H

def match_features(query_img: np.ndarray, train_img: np.ndarray, max_features: int = 500) -> Tuple[np.ndarray, np.ndarray]:
    """Computer matching ORB features between a pair of images.

    Parameters:
        query_img: The first of a pair of images to compute and match features.
        train_img: The second of a pair of images to compute and match features.
        max_features: The maximum number of features to retain.

    Returns:
        pts1: The set of matched point coordinates for the first image.
        pts2: The set of matched point coordinates for the second image.
    """
    query_img_bw = cv2.cvtColor(query_img,cv2.COLOR_BGR2GRAY)
    train_img_bw = cv2.cvtColor(train_img, cv2.COLOR_BGR2GRAY)

    orb = cv2.ORB_create(max_features)

    queryKeypoints, queryDescriptors = orb.detectAndCompute(query_img_bw,None)
    trainKeypoints, trainDescriptors = orb.detectAndCompute(train_img_bw,None)

    matcher = cv2.BFMatcher(crossCheck=True)
    matches = list(matcher.match(queryDescriptors,trainDescriptors) )
    matches.sort(key = lambda x:x.distance)

    pts1 = []
    pts2 = []
    for i in range(8):
        m = matches[i]

        pts2.append(trainKeypoints[m.trainIdx].pt)
        pts1.append(queryKeypoints[m.queryIdx].pt)
    pts1  = np.asarray(pts1)
    pts2 = np.asarray(pts2)

    return (pts1, pts2)

def point_cloud_from_depth(depth: np.ndarray, cam: np.ndarray, color: np.ndarray = None) -> o3d.geometry.PointCloud:
    """Generates a point cloud from a depth map.

    Parameters:
        depth: Depth map to be projected.
        cam: Camera parameters for the depth map viewpoint.
        color: Color vector [R,G,B] to color the point cloud verties.

    Returns:
        The generated point cloud.
    """
    cloud = o3d.geometry.PointCloud()

    # extract camera params
    height, width = depth.shape
    fx = cam[1,0,0]
    fy = cam[1,1,1]
    cx = cam[1,0,2]
    cy = cam[1,1,2]
    intrins = o3d.camera.PinholeCameraIntrinsic(width, height, fx, fy, cx, cy)
    extrins = cam[0]

    # convert deth to o3d.geometry.Image
    depth_image = o3d.geometry.Image(depth)

    cloud = cloud.create_from_depth_image(depth_image, intrins, extrins, depth_scale=1.0, depth_trunc=1000)
    if (colors != None):
        cloud_colors = o3d.utility.Vector3dVector(np.full((len(cloud.points), 3), color))
        cloud.colors = cloud_colors

    return cloud

def reproject(src_depth: np.ndarray, src_cam: np.ndarray, tgt_depth: np.ndarray, tgt_cam: np.ndarray) -> Tuple[  np.ndarray, \
                                                                                                                            np.ndarray, \
                                                                                                                            np.ndarray, \
                                                                                                                            np.ndarray, \
                                                                                                                            np.ndarray] :
    """Computes re-projection indices and depth values between two depth maps.

    Parameters:
        src_depth: Source depth map to be projected.
        src_cam: Source view camera parameters.
        tgt_depth: Target depth map used for re-projection.
        tgt_cam: Target view camera parameters.

    Returns:
        depth_reprojected: Depth map resulting from the re-projection.
        x_reprojected: X-coordinate map resulting from the re-projection.
        y_reprojected: Y-coordinate map resulting from the re-projection.
        x_tgt: X-coordinate map of the projection from source -> target.
        y_tgt: Y-coordinate map of the projection from source -> target.
    """
    height, width = src_depth.shape

    # back-project ref depths to 3D
    x_src, y_src = np.meshgrid(np.arange(0, width), np.arange(0, height))
    x_src, y_src = x_src.reshape([-1]), y_src.reshape([-1])
    xyz_src = np.matmul(np.linalg.inv(src_cam[1,:3,:3]),
                        np.vstack((x_src, y_src, np.ones_like(x_src))) * src_depth.reshape([-1]))

    # transform 3D points from ref to src coords
    xyz_tgt = np.matmul(np.matmul(tgt_cam[0], np.linalg.inv(src_cam[0])),
                        np.vstack((xyz_src, np.ones_like(x_src))))[:3]

    # project src 3D points into pixel coords
    K_xyz_tgt = np.matmul(tgt_cam[1,:3,:3], xyz_tgt)
    xy_tgt = K_xyz_tgt[:2] / K_xyz_tgt[2:3]
    x_tgt = xy_tgt[0].reshape([height, width]).astype(np.float32)
    y_tgt = xy_tgt[1].reshape([height, width]).astype(np.float32)

    # sample the depth values from the src map at each pixel coord
    sampled_depth_tgt = cv2.remap(tgt_depth, x_tgt, y_tgt, interpolation=cv2.INTER_LINEAR)

    # back-project src depths to 3D
    xyz_tgt = np.matmul(np.linalg.inv(tgt_cam[1,:3,:3]),
                        np.vstack((xy_tgt, np.ones_like(x_src))) * sampled_depth_tgt.reshape([-1]))

    # transform 3D points from src to ref coords
    xyz_reprojected = np.matmul(np.matmul(src_cam[0], np.linalg.inv(tgt_cam[0])),
                                np.vstack((xyz_tgt, np.ones_like(x_src))))[:3]

    # extract reprojected depth values
    depth_reprojected = xyz_reprojected[2].reshape([height, width]).astype(np.float32)

    # project ref 3D points into pixel coords
    K_xyz_reprojected = np.matmul(src_cam[1,:3,:3], xyz_reprojected)
    xy_reprojected = K_xyz_reprojected[:2] / (K_xyz_reprojected[2:3] + 1e-7)
    x_reprojected = xy_reprojected[0].reshape([height, width]).astype(np.float32)
    y_reprojected = xy_reprojected[1].reshape([height, width]).astype(np.float32)

    return depth_reprojected, x_reprojected, y_reprojected, x_tgt, y_tgt

def visibility_map(src_depth: np.ndarray, src_cam: np.ndarray, depth_files: List[str], cam_files: List[str], src_index: int = -1, pixel_th: float = 0.1) -> np.ndarray:
    """Computes a visibility map between each the pixels in the source depth map 
        and each depth map in the provided list of target depth map files.

    Parameters:
        src_depth: Source depth map for which the visibility map will be computed.
        src_cam: Source view camera parameters.
        depth_files: File paths for a set of target depth maps.
        cam_files: File paths for a set of corresponding target camera parameters.
        src_index: The index in 'depth_files' corresponding to the source depth.
        pixel_th: Pixel threshold used for binary consensus decisions.

    Returns:
        The visibility map for the source depth map.
    """
    height, width = src_depth.shape
    vis_map = np.not_equal(src_depth, 0.0).astype(np.double)

    for i in range(len(depth_files)):
        if (i==src_index):
            continue

        # get files
        sdf = depth_files[i]
        scf = cam_files[i]

        # load data
        tgt_depth = read_pfm(sdf)
        tgt_cam = read_cam(open(scf,'r'))

        mask = geometric_consistency_mask(src_depth, src_cam, tgt_depth, tgt_cam, pixel_th)
        vis_map += mask

    return vis_map.astype(np.float32)
